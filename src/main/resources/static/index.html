Spring Boot Controller (HomeController.java)

package com.example.demotwo;

import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseBody;

/**
 * Home Controller - Handles routing and API endpoints
 * Separation of concerns: Routing logic isolated from main application class
 */
@Controller
@RequestMapping("/")
public class HomeController {

    // Forward root path to static home page (user experience)
    @GetMapping("/")
    public String redirectToHomePage() {
        return "forward:/index.html";
    }

    // API health check endpoint (monitoring/DevOps)
    @GetMapping("/api/health")
    @ResponseBody
    public String healthCheck() {
        return "OK"; // Simple, standardized health check response
    }

    // Example API endpoint (demonstrates REST design)
    @GetMapping("/indexx")
    @ResponseBody
    public String helloIndex() {
        return "index";
    }
}
